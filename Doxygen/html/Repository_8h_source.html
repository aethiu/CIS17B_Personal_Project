118 void load() {
119 std::ifstream db(db_filename_, std::ios::out | std::ios::binary);
120 if (!db) { throw new db_connection_exception; }
121 db.seekg(0, std::ios::beg);
122 table_.clear();
123
124 // Read header
125 size_t num_rows;
126 db.read(reinterpret_cast<char*>(&num_rows), sizeof(num_rows));
127
128 // Read rows
129 for (size_t i=0; i<num_rows; i++) {
130 load_row(db);
131 }
132 db.close();
133 }

136 void save() const {
137 std::ofstream db(db_filename_, std::ios::out | std::ios::binary);
138 if (!db) { throw new db_connection_exception; }
139 db.seekp(0, std::ios::beg);
140
141 // Write header
142 size_t num_rows = table_.size();
143 db.write(reinterpret_cast<char*>(&num_rows), sizeof(num_rows));
144
145 // Write rows
146 for (const auto& [_, row] : table_) {
147 save_row(db, row);
148 }
149 db.close();
150 }
